  DatabaseFactory? factory,
  String? password,
}) async {
  password ??= 'flutter_sdk';
  if (_instance == null) {

    assert(() {
      if (parseIsWeb) {
        print(

        );
        print(
          'Warning: CoreStoreSembastImp of the Parse_Server_SDK does not encrypt the database on WEB.',
        );
        print(
          '***********************************************************************************************************',
        );
      }
      if (password == 'flutter_sdk') {
